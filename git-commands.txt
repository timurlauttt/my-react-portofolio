# GitHub Commands untuk Deploy
# Jalankan di Command Prompt atau Git Bash, bukan PowerShell

# Navigate to project
cd "d:\2. web\react\my-react-app"

# Initialize git (jika belum)
git init

# Add all files
git add .

# Commit changes
git commit -m "Initial commit - React Portfolio ready for deployment"

# Add remote origin (ganti USERNAME dengan username GitHub Anda)
git remote add origin https://github.com/USERNAME/my-react-portfolio.git

# Push to GitHub
git branch -M main
git push -u origin main
